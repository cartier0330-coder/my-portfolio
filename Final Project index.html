library(WDI)
library(tidyverse)

indicators <- c(
  health_exp_pc = "SH.XPD.CHEX.PC.CD",
  age_dependency = "SP.POP.DPND",
  gdp_pc = "NY.GDP.PCAP.CD",
  doctors_per_1000 = "SH.MED.PHYS.ZS"
)

data <- WDI(country = "all", indicator = indicators, start = 2015, end = 2015, extra = TRUE)

data_clean <- data %>%
  filter(!region %in% "Aggregates") %>%
  drop_na()
model4 <- lm(health_exp_pc ~ age_dependency + gdp_pc + doctors_per_1000, data = data_clean)
summary(model4)
