health_age <- WDI(indicator = c("SH.XPD.CHEX.PC.CD", "SP.POP.DPND.OL"), start = 2015, end = 2015, extra = TRUE)

health_age_clean <- health_age %>%
  filter(!region %in% c("Aggregates")) %>%
  filter(!is.na(SH.XPD.CHEX.PC.CD) & !is.na(SP.POP.DPND.OL))

ggplot(health_age_clean, aes(x = SP.POP.DPND.OL, y = SH.XPD.CHEX.PC.CD, color = region)) +
  geom_point() +
  labs(title = "老人扶養比與健康支出關係（2015）",
       x = "老人扶養比 (%)",
       y = "人均健康支出（美元）") +
  theme_minimal()

model <- lm(SH.XPD.CHEX.PC.CD ~ SP.POP.DPND.OL, data = health_age_clean)
